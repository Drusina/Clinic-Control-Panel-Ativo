-- Clínicas (tenant principal)
CREATE TABLE clinics (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  nome TEXT NOT NULL,
  fantasia TEXT,
  cnpj TEXT UNIQUE NOT NULL,
  razao_social TEXT,
  cidade TEXT,
  uf CHAR(2),
  cep TEXT,
  endereco TEXT,
  cnae TEXT,
  data_abertura DATE,
  capital_social NUMERIC(14,2),
  situacao_receita TEXT,
  -- Contato principal
  responsavel TEXT,
  email TEXT,
  whatsapp TEXT,
  cargo TEXT,
  -- Plano e status
  plano TEXT NOT NULL DEFAULT 'starter' CHECK (plano IN ('starter','pro','enterprise')),
  status TEXT NOT NULL DEFAULT 'prospect' CHECK (status IN ('prospect','proposta','contrato','trial','ativa','suspensa','desativada')),
  suspenso_motivo TEXT,
  suspenso_em TIMESTAMPTZ,
  desativado_em TIMESTAMPTZ,
  retencao_ate DATE, -- Cláusula 14 do contrato (30 dias após desativação)
  etapa INT NOT NULL DEFAULT 1 CHECK (etapa BETWEEN 1 AND 10),
  progresso INT DEFAULT 0 CHECK (progresso BETWEEN 0 AND 100),
  -- Financeiro
  valor_implantacao NUMERIC(12,2),
  valor_recorrente NUMERIC(12,2),
  forma_pagamento TEXT CHECK (forma_pagamento IN ('pix','boleto','cartao','ted')),
  dia_vencimento INT CHECK (dia_vencimento BETWEEN 1 AND 28),
  reajuste_indice TEXT DEFAULT 'IGPM/FGV',
  inicio_recorrencia DATE,
  -- Metadados
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Usuários (referência para auth.users do Supabase)
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  nome TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  whatsapp TEXT,
  avatar_url TEXT,
  is_super_admin BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Vínculo usuário × clínica com role
CREATE TABLE clinic_users (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  clinic_id UUID NOT NULL REFERENCES clinics(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  role TEXT NOT NULL CHECK (role IN ('consultor','gestor','equipe')),
  area TEXT, -- para equipe: Administrativo, Clínico, Atendimento, Marketing, TI
  convite_enviado_em TIMESTAMPTZ,
  aceito_em TIMESTAMPTZ,
  ultimo_acesso TIMESTAMPTZ,
  ativo BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(clinic_id, user_id)
);

-- Histórico de mudanças de status da clínica
CREATE TABLE clinic_status_history (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

  tipo TEXT, -- inadimplencia, solicitacao_cliente, fim_contrato, reestruturacao, outro

-- Timeline de atividades da clínica
CREATE TABLE clinic_activity (

-- Função: obter IDs de clínicas que o usuário pode acessar
CREATE OR REPLACE FUNCTION user_clinic_ids(uid UUID)
RETURNS SETOF UUID
LANGUAGE SQL
SECURITY DEFINER
STABLE
AS $$
  SELECT clinic_id FROM clinic_users WHERE user_id = uid AND ativo = true
  UNION
  SELECT id FROM clinics WHERE EXISTS (SELECT 1 FROM profiles WHERE id = uid AND is_super_admin = true)
$$;

-- Função: verificar se é super_admin
CREATE OR REPLACE FUNCTION is_super_admin()
RETURNS BOOLEAN
LANGUAGE SQL
SECURITY DEFINER
STABLE
AS $$
  SELECT COALESCE((SELECT is_super_admin FROM profiles WHERE id = auth.uid()), false)
$$;

-- Função: role do usuário em uma clínica específica
CREATE OR REPLACE FUNCTION user_role_in_clinic(cid UUID)
RETURNS TEXT
LANGUAGE SQL
SECURITY DEFINER
STABLE
AS $$
  SELECT role FROM clinic_users WHERE clinic_id = cid AND user_id = auth.uid() AND ativo = true LIMIT 1
$$;

-- Trigger: atualizar updated_at
CREATE OR REPLACE FUNCTION touch_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER touch_clinics BEFORE UPDATE ON clinics FOR EACH ROW EXECUTE FUNCTION touch_updated_at();
CREATE TRIGGER touch_acoes BEFORE UPDATE ON acoes FOR EACH ROW EXECUTE FUNCTION touch_updated_at();
CREATE TRIGGER touch_kickoffs BEFORE UPDATE ON kickoffs FOR EACH ROW EXECUTE FUNCTION touch_updated_at();

-- Pilares e perguntas são PÚBLICOS de leitura (referência compartilhada)
-- Apenas super_admin pode editar

-- Policy genérica: SELECT/UPDATE/DELETE baseado em clinic_id do usuário
-- (repetir para todas as tabelas com clinic_id)

-- CLINICS
CREATE POLICY "super_admin_all_clinics" ON clinics FOR ALL USING (is_super_admin()) WITH CHECK (is_super_admin());
CREATE POLICY "user_read_own_clinics" ON clinics FOR SELECT USING (id IN (SELECT user_clinic_ids(auth.uid())));

-- Padrão para tabelas com clinic_id (super_admin vê tudo, usuários veem só suas clínicas)
DO $$
DECLARE
  t TEXT;
  tables TEXT[] := ARRAY[
    'clinic_users','clinic_status_history','clinic_activity','clinic_documents','faturas',
    'kickoffs','perfil_operacional','socios','equipe_interna','parceiros_externos',
    'sistemas_uso','docs_constitutivos','lgpd_termos','assinaturas','diagnosticos',
    'respostas','delegacoes','sub_delegacoes','riscos','acoes','processos','evidencias',
    'documentos_gerais','notificacoes'
  ];
BEGIN
  FOREACH t IN ARRAY tables LOOP
    -- Super admin pode tudo
    EXECUTE format('CREATE POLICY "super_admin_all_%I" ON %I FOR ALL USING (is_super_admin()) WITH CHECK (is_super_admin())', t, t);
    -- Usuários veem só suas clínicas (exceção: profiles, respostas, sub_delegacoes usam join)
    IF t NOT IN ('respostas','sub_delegacoes','notificacoes') THEN
      EXECUTE format('CREATE POLICY "clinic_read_%I" ON %I FOR SELECT USING (clinic_id IN (SELECT user_clinic_ids(auth.uid())))', t, t);
      EXECUTE format('CREATE POLICY "clinic_write_%I" ON %I FOR INSERT WITH CHECK (clinic_id IN (SELECT user_clinic_ids(auth.uid())))', t, t);
      EXECUTE format('CREATE POLICY "clinic_update_%I" ON %I FOR UPDATE USING (clinic_id IN (SELECT user_clinic_ids(auth.uid()))) WITH CHECK (clinic_id IN (SELECT user_clinic_ids(auth.uid())))', t, t);
    END IF;
  END LOOP;
END $$;

-- Profiles: usuário vê o próprio + super_admin vê todos
CREATE POLICY "own_profile" ON profiles FOR ALL USING (id = auth.uid() OR is_super_admin()) WITH CHECK (id = auth.uid() OR is_super_admin());

-- Respostas: via diagnostico_id → clinic
CREATE POLICY "clinic_respostas" ON respostas FOR ALL
  USING (diagnostico_id IN (SELECT id FROM diagnosticos WHERE clinic_id IN (SELECT user_clinic_ids(auth.uid()))))
  WITH CHECK (diagnostico_id IN (SELECT id FROM diagnosticos WHERE clinic_id IN (SELECT user_clinic_ids(auth.uid()))));

-- Sub-delegações: via delegacao_id → clinic
CREATE POLICY "clinic_sub_deleg" ON sub_delegacoes FOR ALL
  USING (delegacao_id IN (SELECT id FROM delegacoes WHERE clinic_id IN (SELECT user_clinic_ids(auth.uid()))))
  WITH CHECK (delegacao_id IN (SELECT id FROM delegacoes WHERE clinic_id IN (SELECT user_clinic_ids(auth.uid()))));

-- Notificações: só as do próprio usuário
CREATE POLICY "own_notifications" ON notificacoes FOR ALL USING (user_id = auth.uid()) WITH CHECK (user_id = auth.uid());
